        self.assertTrue(_np.all(spa == 2))

        self.assertTrue(_np.all(spa == 2))

        self.assertEqual(2, len(spa))
        self.assertTrue(_np.all(spa == 4))

        self.assertEqual(3, len(spa))

        self.assertEqual(3, len(spa))
